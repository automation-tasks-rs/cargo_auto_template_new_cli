86/// all possible library errors for `thiserror`
87#[derive(Error, Debug)]
88pub enum LibraryError {
89    #[error("Name `{0}` is already uppercase.")]
90    Uppercase(String),
91    #[error("Unknown error.")]
92    Unknown,
93}
